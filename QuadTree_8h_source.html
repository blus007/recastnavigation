312 {
313 int curDeep = node->GetDeep();
314 const AABB* aabb = elem->GetAABB();
315 if (!aabb || (!node->IsContain(*aabb) && curDeep > 1))
316 return false;
317 if (atDeep && atDeep == curDeep)
318 {
319 node->AddElem(elem);
320 return true;
321 }
322 float width = node->GetWidth();
323 float height = node->GetHeight();
324 float childWidth = width * 0.5f;
325 float childHeight = height * 0.5f;
326 if (childWidth < aabb->GetWidth() ||
327 childHeight < aabb->GetHeight())
328 {
329 node->AddElem(elem);
330 return true;
331 }
332 if (node->HasChild())
333 {
334 for (int i = 0; i < QT_NODE_COUNT; ++i)
335 {

338 return true;

340 node->AddElem(elem);
341 return true;

432 {
